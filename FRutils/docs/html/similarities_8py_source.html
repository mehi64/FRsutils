2frutil.similarities

5This module provides functions for computing similarities between normalized feature vectors and generating

11Functions

23def _linear_similarity_scalar(v1: float, v2: float) -> float:
24 """
25 Compute linear similarity between two scalar values using the formula:
26 `max(0, 1 - |v1 - v2|)`
27
28 Parameters
29 ----------
30 v1 : float
31 First input value in the range [0.0, 1.0].
32 v2 : float
33 Second input value in the range [0.0, 1.0].
34
35 Returns
36 -------
37 float
38 Similarity score in the range [0.0, 1.0].
39
40 Raises
41 ------
42 ValueError
43 If either input or the output is outside the [0.0, 1.0] range.
44 """
45 sim = max(0.0, 1.0 - abs(v1 - v2))
46 if not ((0.0 <= v1 <= 1.0) and (0.0 <= v2 <= 1.0) and (0.0 <= sim <= 1.0)):
47 raise ValueError("inputs/outputs must be in [0.0, 1.0].")
48 return sim

50def _compute_feature_similarities(x1: np.ndarray, x2: np.ndarray, sim_func) -> np.ndarray:
51 """
52 Compute the similarity between two feature vectors using the provided similarity function.
53
54 Parameters
55 ----------
56 x1 : np.ndarray
57 First feature vector (1D array).
58 x2 : np.ndarray
59 Second feature vector (1D array), must be same length as x1.
60 sim_func : Callable[[float, float], float]
61 Similarity function to apply to each pair of feature values.
62
63 Returns
64 -------
65 np.ndarray
66 1D array of similarity scores between each pair of corresponding features.
67 """
68 fs = np.vectorize(sim_func)(x1, x2)
69 return fs

71def _aggregate_similarities(similarities: np.ndarray, agg_tnorm) -> float:
72 """
73 Aggregate a set of similarity scores into a single value using the provided aggregation tnorm.
74
75 Parameters
76 ----------
77 similarities : np.ndarray
78 Array of similarity scores in the range [0.0, 1.0].
79 agg_tnorm : Callable[[np.ndarray], float]
80 Aggregation tnorm.
81

84 float
85 Aggregated similarity score.

87 Raises

89 ValueError

97def compute_similarity_matrix(X: np.ndarray, sim_func, agg_func) -> np.ndarray:
98 """
99 Compute a pairwise similarity matrix for all instances in a dataset.
100
101 Parameters
102 ----------
103 X : np.ndarray
104 A 2D array of shape (n_samples, n_features).
105 sim_func : Callable[[float, float], float]
106 Similarity function to apply to feature pairs.
107 agg_func : Callable[[np.ndarray], float]
108 Aggregation function for reducing feature similarities to instance similarity.
109
110 Returns
111 -------
112 np.ndarray
113 A 2D similarity matrix of shape (n_samples, n_samples).
114 """
115 n = X.shape[0]
116 sim_matrix = np.zeros((n, n))
117 for i in range(n):
118 sims = np.array([
119 _aggregate_similarities(
120 _compute_feature_similarities(X[i], X[j], sim_func),
121 agg_func
122 ) for j in range(n)
123 ])
124 sim_matrix[i, :] = sims
125 return sim_matrix
