19class SimilarityFunction(ABC):
20 """
21 Abstract base class for scalar similarity functions.
22 """
23 @abstractmethod
24 def compute(self, diff: np.ndarray) -> np.ndarray:
25 pass
26
27 def __call__(self, diff: np.ndarray) -> np.ndarray:
28 return self.compute(diff)

31class LinearSimilarity(SimilarityFunction):
32 """
33 Linear similarity: sim = max(0, 1 - |v1 - v2|)
34 """
35 def compute(self, diff: np.ndarray) -> np.ndarray:
36 return np.maximum(0.0, 1.0 - np.abs(diff))

39class GaussianSimilarity(SimilarityFunction):
40 """
41 Gaussian similarity: sim = exp(-diff^2 / (2 * sigma^2))
42
43 @param sigma: Standard deviation for the Gaussian kernel (must be > 0)
44 """
45 def __init__(self, sigma: float = 0.1):
46 if not (0 < sigma <= 0.5):
47 raise ValueError("sigma must be in the range (0, 0.5]")
48 self.sigma = sigma
49
50 def compute(self, diff: np.ndarray) -> np.ndarray:
51 return np.exp(-(diff ** 2) / (2.0 * self.sigma ** 2))

57def calculate_similarity_matrix(
58 X: np.ndarray,
59 similarity_func: SimilarityFunction,
60 tnorm: Callable[[np.ndarray, np.ndarray], np.ndarray]
61) -> np.ndarray:
62 """
63 Compute the pairwise similarity matrix for samples using a vectorized
64 similarity function and T-norm operator.
65
66 @param X: Normalized input matrix of shape (n_samples, n_features)
67 @param similarity_func: Instance of SimilarityFunction subclass
68 @param tnorm: Callable T-norm function (e.g., min, product, Yager)
69 @return: Similarity matrix of shape (n_samples, n_samples)
70 """
71 n_samples, n_features = X.shape
72 sim_matrix = np.ones((n_samples, n_samples), dtype=np.float64)
73
74 for k in range(n_features):
75 col = X[:, k].reshape(-1, 1) # Shape (n, 1)
76 diff = col - col.T # Shape (n, n)
77 sim_k = similarity_func(diff) # Compute similarity
78 sim_matrix = tnorm(sim_matrix, sim_k) # Apply T-norm
79
80 np.fill_diagonal(sim_matrix, 1.0)
81 return sim_matrix
